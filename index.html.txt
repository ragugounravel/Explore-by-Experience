<script>
const hotels = [
{ name:"Conrad Makkah", location:"Saudi Arabia", experience:"Spiritual", description:"Refined design shaped for serene reflection", price:"BHD 78"},
{ name:"Swissôtel Makkah", location:"Saudi Arabia", experience:"Spiritual", description:"Swiss precision at the crossroads of devotion", price:"BHD 91"},
{ name:"The Lanesborough", location:"United Kingdom", experience:"Architectural", description:"Stately grandeur meets flawless service", price:"BHD 379"},
{ name:"Brown's Hotel", location:"United Kingdom", experience:"Historical", description:"Classic Mayfair elegance with literary soul", price:"BHD 331"},
{ name:"Taj Mahal Palace", location:"India", experience:"Historical", description:"Iconic heritage overlooking Gateway of India", price:"BHD 210"},
{ name:"Burj Al Arab", location:"Saudi Arabia", experience:"Luxury", description:"World-famous ultra luxury landmark stay", price:"BHD 900"}
];

let currentExperience = "";

function loadExperience(type){
currentExperience = type;
document.getElementById("experienceTitle").innerText = type + " Experiences";
document.getElementById("experienceResults").style.display="block";
renderHotels();
window.scrollTo({top:document.getElementById("experienceResults").offsetTop-80,behavior:"smooth"});
}

function renderHotels(country="All"){
const grid = document.getElementById("experienceGrid");
grid.innerHTML="";

let filtered = hotels.filter(h=>h.experience===currentExperience);
if(country!=="All"){
filtered = filtered.filter(h=>h.location===country);
}

filtered.forEach(hotel=>{
grid.innerHTML+=`
<div class="card">
<div class="image-placeholder"></div>
<div class="card-content">
<div class="hotel-name">${hotel.name}</div>
<div class="location">${hotel.location}</div>
<div class="description">${hotel.description}</div>
<div class="price">${hotel.price}</div>
</div>
</div>`;
});
}

function filterCountry(country){
renderHotels(country);
}
</script>
